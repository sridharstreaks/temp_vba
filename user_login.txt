Sub OpenCMDAndPasteCommand()
    Dim cmdCommand As String
    cmdCommand = "echo Hello, this is a test command" ' Change to your desired command

    ' Open CMD
    Shell "cmd.exe", vbNormalFocus
    Application.Wait (Now + TimeValue("00:00:02")) ' Wait for CMD to open

    ' Copy command to clipboard
    Dim objData As Object
    Set objData = CreateObject("MSForms.DataObject")
    objData.SetText cmdCommand
    objData.PutInClipboard

    ' Simulate Ctrl+V to paste
    Application.SendKeys "^v", True
End Sub
