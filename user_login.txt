Sub RunCommandAndReadOutput()
    Dim shell As Object
    Dim execObj As Object
    Dim command As String
    Dim output As String

    ' Set command to execute (replace "ipconfig" with your command)
    command = "cmd /c ipconfig"

    ' Create Shell object
    Set shell = CreateObject("WScript.Shell")
    
    ' Execute command and capture output
    Set execObj = shell.Exec(command)
    
    ' Read output
    output = execObj.StdOut.ReadAll
    
    ' Display the output
    MsgBox output, vbInformation, "Command Output"

    ' Clean up
    Set execObj = Nothing
    Set shell = Nothing
End Sub
